### 1. 配置模块

**概念定义**：集中配置管理——将 API 密钥、模型参数、文件路径等可配项集中到 `Config` 数据类中。一处修改全局生效，环境切换只需改 `.env` 文件，代码零改动。

```python
"""
config.py
配置模块
"""
import os
from dataclasses import dataclass
from dotenv import load_dotenv

load_dotenv()


@dataclass
class Config:
    """应用配置"""
    # OpenAI
    OPENAI_API_KEY: str = os.getenv("OPENAI_API_KEY", "")
    LLM_MODEL: str = "gpt-4o-mini"
    EMBEDDING_MODEL: str = "text-embedding-3-small"

    # 向量库
    CHROMA_PERSIST_DIR: str = "./chroma_db"
    COLLECTION_NAME: str = "knowledge_base"

    # 分块
    CHUNK_SIZE: int = 500
    CHUNK_OVERLAP: int = 50

    # 检索
    RETRIEVAL_K: int = 4

    # 文档目录
    DOCUMENTS_DIR: str = "./data/documents"


config = Config()
```

### 2. 文档加载

**核心定位**：文档加载器作为统一的文件解析入口——封装 PDF、Word、Markdown 等不同格式的加载逻辑，对外暴露一致接口。新增文件格式只需注册对应加载器，业务代码无需改动。

```python
"""
document_loader.py
文档加载模块
"""
from pathlib import Path
from typing import List
from langchain.schema import Document
from langchain_community.document_loaders import (
    TextLoader,
    PyPDFLoader,
    Docx2txtLoader,
    UnstructuredMarkdownLoader,
    DirectoryLoader,
)


class DocumentLoader:
    """文档加载器"""

    LOADERS = {
        ".txt": TextLoader,
        ".pdf": PyPDFLoader,
        ".docx": Docx2txtLoader,
        ".md": UnstructuredMarkdownLoader,
    }

    def load_file(self, file_path: str) -> List[Document]:
        """加载单个文件"""
        path = Path(file_path)
        ext = path.suffix.lower()

        if ext not in self.LOADERS:
            raise ValueError(f"不支持的文件类型: {ext}")

        loader_cls = self.LOADERS[ext]
        loader = loader_cls(str(path))
        documents = loader.load()

        # 添加来源元数据
        for doc in documents:
            doc.metadata["source"] = path.name
            doc.metadata["file_type"] = ext

        return documents

    def load_directory(self, dir_path: str) -> List[Document]:
        """加载目录下所有文件"""
        all_documents = []
        path = Path(dir_path)

        for ext in self.LOADERS.keys():
            for file_path in path.glob(f"**/*{ext}"):
                try:
                    docs = self.load_file(str(file_path))
                    all_documents.extend(docs)
                    print(f"  加载: {file_path.name} ({len(docs)} 个文档)")
                except Exception as e:
                    print(f"  跳过: {file_path.name} - {e}")

        return all_documents
```

### 3. 向量存储

**概念定义**：向量存储是 RAG 系统的知识中枢——使用 Embedding 模型将文本转换为高维向量，存入 Chroma 向量数据库。检索时通过语义相似度计算召回最相关内容，而非依赖关键词匹配。

```python
"""
vectorstore.py
向量存储模块
"""
from typing import List, Optional
from langchain.schema import Document
from langchain_community.vectorstores import Chroma
from langchain_openai import OpenAIEmbeddings
from langchain.text_splitter import RecursiveCharacterTextSplitter

from .config import config


class VectorStoreManager:
    """向量存储管理器"""

    def __init__(self):
        self.embeddings = OpenAIEmbeddings(
            model=config.EMBEDDING_MODEL
        )
        self.text_splitter = RecursiveCharacterTextSplitter(
            chunk_size=config.CHUNK_SIZE,
            chunk_overlap=config.CHUNK_OVERLAP
        )
        self._vectorstore: Optional[Chroma] = None

    @property
    def vectorstore(self) -> Chroma:
        """获取向量存储"""
        if self._vectorstore is None:
            self._vectorstore = Chroma(
                collection_name=config.COLLECTION_NAME,
                embedding_function=self.embeddings,
                persist_directory=config.CHROMA_PERSIST_DIR
            )
        return self._vectorstore

    def add_documents(self, documents: List[Document]) -> int:
        """添加文档"""
        # 分块
        chunks = self.text_splitter.split_documents(documents)
        print(f"分块: {len(documents)} 个文档 -> {len(chunks)} 个块")

        # 添加到向量库
        self.vectorstore.add_documents(chunks)

        return len(chunks)

    def search(self, query: str, k: int = None) -> List[Document]:
        """搜索"""
        k = k or config.RETRIEVAL_K
        return self.vectorstore.similarity_search(query, k=k)

    def get_retriever(self):
        """获取检索器"""
        return self.vectorstore.as_retriever(
            search_kwargs={"k": config.RETRIEVAL_K}
        )

    def get_stats(self) -> dict:
        """获取统计信息"""
        return {
            "total_documents": self.vectorstore._collection.count()
        }
```

### 4. 问答链

**核心定位**：RAG（检索增强生成）模式——先检索相关知识片段作为上下文，再让 LLM 基于检索结果回答。有效解决 LLM 知识截止日期问题和事实幻觉，是实现企业私域知识问答的核心架构。

```python
"""
qa_chain.py
问答链模块
"""
from typing import List, Tuple
from langchain_openai import ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder
from langchain_core.messages import HumanMessage, AIMessage
from langchain_core.output_parsers import StrOutputParser
from langchain_core.runnables import RunnablePassthrough

from .config import config
from .vectorstore import VectorStoreManager


class QASystem:
    """问答系统"""

    def __init__(self):
        self.llm = ChatOpenAI(
            model=config.LLM_MODEL,
            temperature=0
        )
        self.vectorstore_manager = VectorStoreManager()
        self.chat_history: List[Tuple[str, str]] = []

        # RAG Prompt
        self.qa_prompt = ChatPromptTemplate.from_messages([
            ("system", """你是一个知识库问答助手。基于提供的上下文回答问题。

规则：
1. 只使用上下文中的信息回答
2. 如果上下文不足以回答，明确说明
3. 回答时引用来源（如：根据[文档名]...）
4. 保持回答简洁准确

上下文：
{context}"""),
            MessagesPlaceholder(variable_name="chat_history"),
            ("human", "{question}")
        ])

    def _format_docs(self, docs) -> str:
        """格式化文档"""
        formatted = []
        for i, doc in enumerate(docs, 1):
            source = doc.metadata.get("source", "未知")
            formatted.append(f"[{i}] 来源: {source}\n{doc.page_content}")
        return "\n\n".join(formatted)

    def _get_chat_history(self):
        """获取对话历史"""
        messages = []
        for human, ai in self.chat_history[-5:]:  # 保留最近5轮
            messages.append(HumanMessage(content=human))
            messages.append(AIMessage(content=ai))
        return messages

    def ask(self, question: str) -> dict:
        """问答"""
        # 检索
        retriever = self.vectorstore_manager.get_retriever()
        docs = retriever.invoke(question)

        # 构建链
        chain = (
            {
                "context": lambda x: self._format_docs(docs),
                "chat_history": lambda x: self._get_chat_history(),
                "question": RunnablePassthrough()
            }
            | self.qa_prompt
            | self.llm
            | StrOutputParser()
        )

        # 生成回答
        answer = chain.invoke(question)

        # 更新历史
        self.chat_history.append((question, answer))

        # 返回结果
        return {
            "question": question,
            "answer": answer,
            "sources": [
                {
                    "content": doc.page_content[:200] + "...",
                    "source": doc.metadata.get("source", "未知")
                }
                for doc in docs
            ]
        }

    def clear_history(self):
        """清除对话历史"""
        self.chat_history = []
```

### 5. Web API

**概念定义**：API 层将内部能力暴露为 RESTful 接口，使得前端应用、移动端、第三方系统均可调用问答能力。FastAPI 通过 Pydantic 模型自动校验请求和响应，实现系统解耦与标准化集成。

```python
"""
api.py
Web API
"""
from fastapi import FastAPI, HTTPException, UploadFile, File
from pydantic import BaseModel
from typing import List, Optional
import shutil
from pathlib import Path

from .config import config
from .document_loader import DocumentLoader
from .vectorstore import VectorStoreManager
from .qa_chain import QASystem

app = FastAPI(title="知识库问答系统")

# 全局实例
qa_system = QASystem()
doc_loader = DocumentLoader()
vs_manager = VectorStoreManager()


class QuestionRequest(BaseModel):
    question: str


class AnswerResponse(BaseModel):
    question: str
    answer: str
    sources: List[dict]


@app.post("/ask", response_model=AnswerResponse)
async def ask_question(request: QuestionRequest):
    """问答接口"""
    try:
        result = qa_system.ask(request.question)
        return AnswerResponse(**result)
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))


@app.post("/upload")
async def upload_document(file: UploadFile = File(...)):
    """上传文档"""
    # 保存文件
    save_path = Path(config.DOCUMENTS_DIR) / file.filename
    save_path.parent.mkdir(parents=True, exist_ok=True)

    with open(save_path, "wb") as f:
        shutil.copyfileobj(file.file, f)

    # 加载并索引
    try:
        docs = doc_loader.load_file(str(save_path))
        count = vs_manager.add_documents(docs)
        return {"message": f"成功索引 {count} 个文档块", "file": file.filename}
    except Exception as e:
        raise HTTPException(status_code=400, detail=str(e))


@app.post("/index")
async def index_directory():
    """索引文档目录"""
    docs = doc_loader.load_directory(config.DOCUMENTS_DIR)
    if not docs:
        return {"message": "没有找到文档"}

    count = vs_manager.add_documents(docs)
    return {"message": f"成功索引 {count} 个文档块"}


@app.get("/stats")
async def get_stats():
    """获取统计信息"""
    return vs_manager.get_stats()


@app.post("/clear-history")
async def clear_history():
    """清除对话历史"""
    qa_system.clear_history()
    return {"message": "对话历史已清除"}
```

## 常见踩坑

1. **Embedding 模型与检索质量不匹配**：生产环境中切换 Embedding 模型（如从 text-embedding-3-small 换为 bge-large-zh）后，Chroma 中已有向量的维度与新模型不匹配，需重建索引。切换前务必清空持久化目录并重新索引所有文档。

2. **文档分块策略选择不当**：chunk_size=500、chunk_overlap=50 并非通用最优值。代码文档建议用 1000+200 以保留完整函数，合同条款建议用 300+30 以精确匹配。调整后务必在测试样本集上验证检索召回率，避免盲目套用。

3. **Chroma 持久化目录并发冲突**：多进程或容器多副本同时写入同一个 `persist_directory` 会导致数据库损坏。生产部署时每个实例使用独立目录，或改用 PGVector、Milvus 等支持并发读写的向量数据库。

4. **多轮对话 Token 超限**：代码按轮次保留最近 5 轮，但未做 Token 计数。若某一轮回答超长（如 4000 Token），仍可能超出 LLM 上下文窗口。建议改用 `tiktoken` 按 Token 数截断，保留策略为：总 Token 数不超过模型上下文窗口的 50%。

5. **FastAPI 接口缺乏鉴权**：`/ask` 和 `/upload` 接口直接暴露，生产环境易被恶意调用。必须添加 API Key 鉴权（中间件或依赖注入）和速率限制，建议集成 `slowapi` 实现基于 IP 或 Token 的限流。

## 课后练习

1. 增加 PDF 表格解析能力：调研并集成 `pdfplumber` 或 `Camelot`，使系统能提取 PDF 中的表格数据并存入向量库，支持针对表格内容的问答。

2. 实现混合检索（稠密 + 稀疏）：在语义向量检索之外增加 BM25 关键词检索，将两者结果按加权分数（如 0.7 语义 + 0.3 关键词）融合排序，提升专业术语查询的命中率。

3. 添加对话历史持久化：将 `chat_history` 存入 SQLite 或 Redis，使服务重启后多轮对话不丢失。设计一个 `ConversationStore` 类封装持久化逻辑。

4. 使用 Docker 容器化部署：编写 `Dockerfile` 和 `docker-compose.yml`，将问答系统打包为容器镜像，配置环境变量注入和卷挂载，实现一键部署。
